\section{Keys}

\DescribeOption{title}
\DescribeOption{labnum}
\DescribeOption{course}
Set the title of the report, the lab number, and the course name respectively. \texttt{course} is only used on the title page. \texttt{labnum} is used as a prefix for section numbering, as well as in the header of each page along with \texttt{title}.\\

\DescribeOption{author}
\DescribeOption{partner}
Sets the author and lab partner of the paper. The \texttt{partner} key is optional, when omitted all references to the partner will be hidden. The author and partner are listed on the title page, as well as in the header of each page.\\

\section{Macros}
\begin{function}{\task}
  \begin{syntax}
    \cs{task}\oarg{name}\marg{number}
  \end{syntax}
  Inserts a new page, sets \cs{thesection} to \marg{number}, and calls \cs{section}\marg{name}.
\end{function}

\begin{function}{\subtask}
  \begin{syntax}
    \cs{subtask}\oarg{name}\marg{number}
  \end{syntax}
  Sets \cs{thesubsection} to \marg{number} and calls \cs{subsection}\marg{name}.
\end{function}

\begin{function}{\objective}
  % \begin{syntax}
  %   \cs{objective}
  % \end{syntax}
  Calls \verb|\paragraph{Objective}|.
\end{function}

\begin{function}{\conclusion}
  % \begin{syntax}
  %   \cs{conclusion}
  % \end{syntax}
  Calls \verb|\paragraph{Conclusion}|.
\end{function}

\begin{function}{\inputcode}
  \begin{syntax}
    \cs{inputcode}\oarg{listings opts}\marg{filename}
  \end{syntax}
  Calls \cs{lstinputlisting}\oarg{listings opts}\marg{filename}. Recommended to set the code language using the listings options. Default code language is Python. Sets the title of the created \texttt{listings} environment to \marg{filename}.
\end{function}

\begin{function}{\listofplots}
  % \begin{syntax}
  %   \cs{listofplots}
  % \end{syntax}
  Creates a list of plots using the \verb|tocloft| package.
\end{function}

\begin{function}{\listofcircuits}
  % \begin{syntax}
  %   \cs{listofcircuits}
  % \end{syntax}
  Creates a list of circuits using the \verb|tocloft| package.
\end{function}


\section{Environments}
\begin{function}{plot}
  \begin{syntax}
    \cs{begin}\marg{plot}
  \end{syntax}
  Wrapper for a centered \verb|tikzpicture|. If unstarred, increments \cs{theplots} and adds an entry to the \cs{listofplots}. Support for \cs{caption}\oarg{short list entry}\marg{caption text} and \cs{label}\marg{label name}. Sane naming when used with \cs{autoref}\marg{target}. Note that, when using \verb|pgfplots|, the \cs{caption} and \cs{label} commands must be outside the \verb|axis| environment.
\end{function}

\begin{function}{circuit}
  \begin{syntax}
    \cs{begin}\marg{circuit}
  \end{syntax}
  Wrapper for a centered \verb|circuitikz|. If unstarred, increments \cs{thecircuits} and adds an entry to the \cs{listofcircuits}. Support for \cs{caption}\oarg{short list entry}\marg{caption text} and \cs{label}\marg{label name}. Sane naming when used with \cs{autoref}\marg{target}.
\end{function}
